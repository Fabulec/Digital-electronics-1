# 2.D latch:
## VHDL code listing of the process ```p_d_latch```:

```vhdl
entity d_latch1 is
    Port (
            en    : in STD_LOGIC;
            arst  : in STD_LOGIC;
            d     : in STD_LOGIC;
            q     : out STD_LOGIC;
            q_bar : out STD_LOGIC);
end d_latch1;

architecture Behavioral of d_latch1 is

begin
------------------------------------------------------------------------
-- p_alarm:
-- A combinational process of alarm clock.
------------------------------------------------------------------------
p_d_latch : process (d, arst, en)

begin
    if (arst = '1' ) then
        q      <= '0';
        q_bar  <= '1';
        
   elsif (en = '1') then
         q       <= d;
        q_bar    <= not d;
    
    end if ;
    
end process p_d_latch;



end Behavioral;

```

### VHDL code listing of the process ```p_jk_ff_rst```
```vhdl
entity jk_ff_rst is
Port (                            
           clk    : in STD_LOGIC; 
           rst  : in STD_LOGIC;  
           j     : in STD_LOGIC;
           k     : in STD_LOGIC; 
           q     : out STD_LOGIC; 
           q_bar : out STD_LOGIC);
          
          
end jk_ff_rst;

architecture Behavioral of jk_ff_rst is
signal s_q : std_logic;
begin

p_d_latch : process (clk)

begin
if rising_edge(clk) then
    if(rst = '1') then
        s_q <= '0';
    else 
        if (j = '0' and k = '0' ) then 
                s_q <= s_q;
                
        elsif (j = '0' and k = '1' ) then
                s_q <= '0' ;
   
        elsif (j = '1' and k = '0' ) then
                s_q <= '1' ;
   
        elsif (j = '1' and k = '1' ) then
                s_q <= not s_q ;
   
   
   
        end if;
        
    end if;
    
 end if;

    
end process p_d_latch;

q       <= s_q;
q_bar   <= not s_q;

end Behavioral;
```
